#include <iostream>
#include <locale>
//Escreva um programa para exibir os nÃºmeros contidos no intervalo de 50 a 10, inclusive.
using namespace std;

int main()
{
    setlocale(LC_ALL, "Portuguese");

    int contador;

    contador = 50;
    while (contador >= 10) {
        cout<< contador<<endl;
        contador= contador - 1;
    }
    return 0;
}
